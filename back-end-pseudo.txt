import os
import hashlib

"""
:: passHash - Hashes user passwords
Param pw: Password to put into hash form with true-random salt
Returns hashed password with true-random salt
"""
def passHash(pw):
	salt = os.urandom(64)
	hashed = hashlib.pbkdf2_hmac('sha256', pw.encode('utf-8'), salt, 1000000)
	return (salt + hashed)
